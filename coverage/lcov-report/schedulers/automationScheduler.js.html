const cron = require('node-cron');
const AutomationService = require('../services/automationService');
const AutomatedFeeNotificationsService = require('../services/automatedFeeNotificationsService');
const AutomatedAttendanceNotificationsService = require('../services/automatedAttendanceNotificationsService');
const WhatsAppService = require('../services/whatsappService');
 
class AutomationScheduler {
  constructor() {
    this.automationService = new AutomationService();
    this.feeNotificationsService = new AutomatedFeeNotificationsService();
    this.attendanceNotificationsService = new AutomatedAttendanceNotificationsService();
    this.whatsappService = new WhatsAppService();
    this.schedulerRunning = false;
    this.cronJobs = [];
  }
 
  /**
   * Start the automation scheduler
   */
  async start() {
    if (this.schedulerRunning) {
      console.log('‚ö†Ô∏è Automation scheduler is already running');
      return;
    }
 
    console.log('üöÄ Starting automation scheduler...');
 
    try {
      // Start the automation service first
      await this.automationService.start();
 
      // Schedule legacy automation rule checks every minute
      this.scheduleAutomationChecks();
 
      // Schedule Phase 6 automation services
      this.scheduleFeeReminders();
      this.scheduleAttendanceNotifications();
 
      // Schedule health checks every 5 minutes
      this.scheduleHealthChecks();
 
      // Schedule cleanup tasks daily at 2 AM
      this.scheduleCleanupTasks();
 
      this.schedulerRunning = true;
      console.log('‚úÖ Automation scheduler started successfully with Phase 6 services');
 
    } catch (error) {
      console.error('‚ùå Failed to start automation scheduler:', error.message);
      throw error;
    }
  }
 
  /**
   * Stop the automation scheduler
   */
  async stop() {
    if (!this.schedulerRunning) {
      console.log('‚ö†Ô∏è Automation scheduler is not running');
      return;
    }
 
    console.log('üõë Stopping automation scheduler...');
 
    try {
      // Stop all cron jobs
      this.cronJobs.forEach(job => {
        if (job && job.stop) {
          job.stop();
        }
      });
      this.cronJobs = [];
 
      // Stop the automation service
      await this.automationService.stop();
 
      this.schedulerRunning = false;
      console.log('‚úÖ Automation scheduler stopped successfully');
 
    } catch (error) {
      console.error('‚ùå Error stopping automation scheduler:', error.message);
    }
  }
 
  /**
   * Schedule automation rule checks (legacy)
   */
  scheduleAutomationChecks() {
    console.log('‚è∞ Scheduling legacy automation rule checks (every minute)...');
 
    const job = cron.schedule('* * * * *', async () => {
      try {
        await this.automationService.checkAndExecuteRules();
      } catch (error) {
        console.error('‚ùå Legacy automation check failed:', error.message);
      }
    }, {
      scheduled: true,
      timezone: process.env.TZ || 'Asia/Karachi'
    });
 
    this.cronJobs.push(job);
    console.log('‚úÖ Legacy automation rule checks scheduled');
  }
 
  /**
   * Schedule fee reminder checks
   */
  scheduleFeeReminders() {
    console.log('üí∞ Scheduling fee reminder checks (every hour at minute 5)...');
 
    const job = cron.schedule('5 * * * *', async () => {
      try {
        console.log('üîî Running automated fee reminder check...');
        const result = await this.feeNotificationsService.checkAndSendDueDateReminders();
        console.log(`üìä Fee reminders completed: ${result.totalReminders} reminders sent to ${result.totalOrganizations} organizations`);
      } catch (error) {
        console.error('‚ùå Fee reminder check failed:', error.message);
      }
    }, {
      scheduled: true,
      timezone: process.env.TZ || 'Asia/Karachi'
    });
 
    this.cronJobs.push(job);
    console.log('‚úÖ Fee reminder checks scheduled');
  }
 
  /**
   * Schedule attendance notification checks
   */
  scheduleAttendanceNotifications() {
    console.log('üìö Scheduling attendance notification checks (every 15 minutes during school hours)...');
 
    const job = cron.schedule('*/15 * * * *', async () => {
      try {
        // Only run during typical school hours (8 AM to 6 PM)
        const now = new Date();
        const hour = now.getHours();
        
        if (hour >= 8 && hour <= 18) {
          console.log('üéì Running automated attendance notification check...');
          const result = await this.attendanceNotificationsService.checkAndProcessAttendanceNotifications();
          console.log(`üìä Attendance notifications completed: ${result.totalNotifications} notifications sent to ${result.totalOrganizations} organizations`);
        } else {
          console.log('üåô Outside school hours, skipping attendance notifications');
        }
      } catch (error) {
        console.error('‚ùå Attendance notification check failed:', error.message);
      }
    }, {
      scheduled: true,
      timezone: process.env.TZ || 'Asia/Karachi'
    });
 
    this.cronJobs.push(job);
    console.log('‚úÖ Attendance notification checks scheduled');
  }

   * Schedule health checks

  }

    try {

      console.log(`  - Redis Queue: ${status.queue?.connected ? '‚úÖ Connected' : '‚ùå Disconnected'}`);
      console.log(`  - Rate Limiting: ${status.rateLimit?.connected ? '‚úÖ Connected' : '‚ùå Disconnected'}`);
      console.log(`  - Active WhatsApp Sessions: ${whatsappHealth.activeSessions || 0}`);
 
      // Phase 6 services health
      console.log('üìä Phase 6 Services Status:');
      console.log(`  - Fee Notifications Service: ‚úÖ Loaded`);
      console.log(`  - Attendance Notifications Service: ‚úÖ Loaded`);
 
      // Check for critical issues
      if (!status.running) {
        console.warn('‚ö†Ô∏è Legacy automation service is not running - attempting restart...');
        await this.restartAutomationService();
      }
 
      if (whatsappHealth.status === 'unhealthy') {
        console.warn('‚ö†Ô∏è WhatsApp service is unhealthy:', whatsappHealth.error);

      if (!status.wppconnect?.connected) {
        console.warn('‚ö†Ô∏è WPPConnect is not connected - this may affect message delivery');
      }
 
      if (!status.firebase?.connected) {
        console.warn('‚ö†Ô∏è Firebase is not connected - this may affect data operations');
      }
 
      console.log('‚úÖ Health check completed');
 
    } catch (error) {
      console.error('‚ùå Health check failed:', error.message);
    }
  }
